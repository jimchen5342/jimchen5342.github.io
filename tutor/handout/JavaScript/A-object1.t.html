實字模式(literal notation patterns)

：當只想為物件建立一個執行個體時，會使用此方式； 此方式不適合於需建立多個相同物件模組的運用。 
		
			let person = {};       // 宣告方式一
			person.name = "John";
			person.age = 25;
			person.show = function(){
				return this.name + " is " + this.age;
			};
			write(person.show());

			let student = {         // 宣告方式二
				name: "Mary",
				age: 20,
				show: function(){
					return this.name + " is " + this.age;
				}
			}; 
			write(student.show());
