賦值運算子
名稱 簡化的運算子 意義
賦值 x = y x = y
加法賦值 x += y x = x + y
減法賦值 x -= y x = x - y
乘法賦值 x *= y x = x * y
除法賦值 x /= y x = x / y

解構賦值(ES6)：用於快速提取(extract)陣列或物件中的資料 
		
				let arr = [1, 2, 3, 4, 5]; 
				let [y, z] = arr;                   // 陣列解構
				document.write("陣列解構 x: " + y);  // 1 
				document.write("陣列解構 y: " + z);  // 2

				// 將剩餘部分解構到一個變數
				let [a, ...b] = [1, 2, 3]; 
				document.write("陣列解構 a: " + a);  // 1 
				document.write("陣列解構 b: " + b);  // [2, 3]
			 
				let o = {p: 42, q: true, r: '', s: ''}; 
				let {p, q, ...obj} = o;            // 物件解構
				document.write("物件解構 p: " + p);  // 42 
				document.write("物件解構 q: " + q);  // true
				document.write("物件解構 obj: " + 
				  JSON.stringify(obj));            // {r: '', s: ''}
