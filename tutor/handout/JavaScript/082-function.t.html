遞迴函式(recursion function)

：在一個函式當中再去呼叫它自己， 在寫遞回函式的時候一定記得設定終止條件，否則很容易跳不出來，導致無窮迴圈的情況產生。 
			
				var x = 0; 
				function recursion(i){ 
					x += i; 
					document.write(i + ": " + x + "<br />");
					if(i < 10)
						recursion(i + 1); 
				} 
				recursion(1);

匿名函式(anonymous function)

			
				// 第一種寫法
				setTimeout(function(){ 
					//函式內容 
				}, 600);

				// 第二種寫法
				let loop = function(){ 
					//函式內容 
				}
				setTimeout(loop, 600);

				// 第二種寫法，補充
				function cycle(callback){ 
					if(typeof callback == "function")
						callback();
				}
				cycle(loop)

立即函式(Immediately Invoked Function Expression，IIFE)：

js 檔載入後即會執行；主要在於產生一個私有的 scope，使在其內宣告的變數不會與外界混淆； 很常被使用來進行只會被執行一次的程式碼，譬如說程式的初始化動作。 
			
				// 第一種寫法
				(function () { 
					//函式內容 
				})();

				// 第二種寫法
				(function(window, undefined) {
					//函式內容
				})(window);

巢狀函式(nest function)

：函式中的語句中，還可以包含其他的函式，這稱為內部函式(inner)，或是巢狀函式(nested)的結構。 
			
				function query(arg){
					 function response(){
						return "hi, ";
					}

					return response() + arg;
				}

				let x = query("John");
				document.write(x);

閉包(closure)

：是函式的組合，以及該宣告函式所包含的作用域環境（lexical environment）。 閉包實用之處，在於能讓你把一些資料（透過作用域環境）與操控這些資料的函式相關聯。 很明顯地，這與把一些資料（物件屬性）與一些方法的相關聯的物件導向程式設計（object-oriented programming）相似。 參考自 MDN 
			
				// 回傳 function
				function f1(x){
					let a = 10;          // 私有變數

					return function(){ 
						return x + a; 
					}; // 注意要有分號
				}
				let f2 = f1(1);
				document.write("f1: " + f2() + "<br />");

				// 回傳 object
				function f3(x){
					let z = 0;           // 私有變數
					function sum(a, b){  // 私有函式(巢狀)
						z += a + b;
						return z;
					}
					return {
						name: "f3",
						x: x,
						z: z,
						add: function(y){
							return sum(x, y);
						}
					}; // 注意要有分號
				}
				let f4 = f3(10); 
				document.write("function: " + f4.name + "<br />");
				document.write("x: " + f4.x + "<br />");
				document.write("add: " + f4.add(20) + "<br />");
				document.write("add: " + f4.add(1) + "<br />");

箭頭函式(arrow function)

：ES6 
			
				// 第一種：無參數
				let f1 = () => {
					document.write(10 * 20);
				}

				// 第二種：只有一個參數
				let f2 = a => {
					 document.write(a + 5);
				}

				// 第三種：多個參數
				let f3 = (a, b) => {
					document.write(a + 5);
				}

				// 範例
				let x = 0;
				setInterval(()=>{
					 document.write("x: " + x++ + "<br />");
				}, 1000);

Promise

			
				function f1(a, b){
					return new Promise((success, error) => {
						if(a > b)
							success();
						else
							error("Error: a <= b");
					})
				}

				function f2(x, y){ 
					return new Promise((success, error) => { 
						if(x != y) 
							success(); 
						else 
							error('Error: x == y'); 
					}) 
				}

				f1(10, 2) 
				.then(()=>{ 
					return f2(5, 2); 
				})
				.then(()=>{
					alert("完成......")
				})				
				.catch(err=>{
					document.write(err); 
				});
  1. call by value(傳值)： Boolean, String, Number, null, undefined

				let x = ["apple", "orange"], y = x;

javascript 在呼叫每個函數時，都會將引數放到一個稱為 arguments 的變數裏， arguments 是一個類似陣列形態，我們可以透過 arguments 來存取每一個參數 可透過 arguments 作到多載(overload)的功能 
			
				function add(){
					let x = 0;
					for(let i = 0; i < arguments.length; i++){
						x += arguments[i];
					}
					return x;
				}
				let a = add(1, 2, 3, 4, 5);
				document.write(a);
			
		
			
				function Person(name) {
					this.name = name;
					
					this.sayName = function() {
						console.log("Hello my name is: ", this.name);
						return this;
					};
				
					this.changeName = function(name) {
						this.name = name;
						return this;
					};
				}
				
				let person = new Person("John");
				person.sayName().changeName("Timmy").sayName();
