宣告

  1. function

    ：宣告
  2. function name：宣告 / 呼叫

  3. ()

    ：引數(argument, 宣告) / 參數(parameter, 呼叫)；

    ,

    ：分隔引數/參數

  4. {}

    ：程式區塊

  5. return

    ：傳回值
			
				function add(a, b) { //函數宣告式
					return a + b;
				}

				let substract = function (a, b) { //匿名函式
					return a - b;
				}

				function calculate(callback) { // 回調函式(callback)
					return callback(100, 50);
				}

				let x = add(5, 20); // 呼叫函式
				let y = calculate(add); // callback
				let z = calculate(substract); // callback
				document.write("x: " + x + ", y: " + y + ", z: " + z);
主要在於產生一個私有的 scope，使在其內宣告的變數不會與外界混淆 
			
				(function () { 
					//函式內容 
				})();
註：立即函式在 js 檔，所有的函式內容都要在區塊內部

巢狀函式(nest function)

			
				function query(arg){
					 function response(){
						return "hi, ";
					}

					return response() + arg;
				}

				let x = query("jim");
				document.write(x);

遞迴函式(recursion function)

			
				var x = 0; 
				function recursion(i){ 
					x += i; 
					document.write(x + "<br />");
					if(i < 10)
						recursion(i + 1); 
				} 
				recursion(1);

閉包函式(closure)

			
				function f1(x){ 
					return function(){ 
						return x + 10; 
					}; 
				} 
				document.write(f1(1)());

javascript 在呼叫每個函數時，都會將引數放到一個稱為 arguments 的變數裏，arguments 是一個類似陣列形態，我們可以透過 arguments 來存取每一個參數 
			
				function add(){
					let x = 0;
					for(let i = 0; i < arguments.length; i++){
						x += arguments[i];
					}
					return x;
				}
				let a = add(1, 2, 3, 4, 5);
				document.write(a);
