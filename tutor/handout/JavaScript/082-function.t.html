巢狀函式(nest function)

			
				function query(arg){
					 function response(){
						return "hi, ";
					}

					return response() + arg;
				}

				let x = query("jim");
				document.write(x);

遞迴函式(recursion function)

			
				var x = 0; 
				function recursion(i){ 
					x += i; 
					document.write(x + "<br />");
					if(i < 10)
						recursion(i + 1); 
				} 
				recursion(1);
  1. call by value(傳值)： Boolean, String, Number, null, undefined
  2. call by reference(傳參考)：
			
				let a = 2, b = a;
				a = 10;
				document.write("b: " + b);

				let f1 = ["apple", "orange"], f2 = f1;
				f1[0] = "fish";
				document.write("f2[0]: " + f2[0]);
