物件實字(object literal)

：當只想為物件建立一個執行個體時，會使用此方式； 此方式不適合於需建立多個相同物件模組的運用。 
		
			let person = new Object(); // 方式一
			person.name = "John";
			person.age = 25; 
			person.show = function(){ 
				return this.name + " is " + this.age; 
			} 
			write(person.show());

			let student = { 
				name: "Mary", 
				age: 25, 
				show: function(){ 
					return this.name + " is " + this.age; 
				} 
			}

			let student1 = new Object(student);    // 方式二
			write(student1.show());

			let student2 = Object.create(student); // 方式三
			write(student2.show());
