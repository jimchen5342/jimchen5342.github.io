物件實字(object literal)

：當只想為物件建立一個執行個體時，會使用此方式； 此方式不適合於需建立多個相同物件模組的運用。 
		
			let person = new Object();
			person.name = "John";
			person.age = 25; 
			person.toString = function(){ 
				return this.name + " is " + this.age; 
			} 
			write(person.toString());

			let person = { 
				name: "John", 
				age: 25, 
				toString: 
				function(){ 
					return this.name + " is " + this.age; 
				} 
			}
			let student = new Object(person);
			let student = Object.create(person);
