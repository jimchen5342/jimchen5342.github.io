函式建構式

：函式建構式建立了函式實例，不過每次呼叫建構式時，都會建立一次函式實例； 會造成記憶體浪費。 
		
			function Person(name, age) {      // 宣告
				this.name = name;               // 公開成員變數
				let profile = this.name;        // 私有成員變數

				function getAge() {             // 私有成員函數 
					return age;
				}
				this.getProfile = function() {  // 公開成員函數 
					return profile + " is " + getAge();
				} 
			}
			let person = new Person("john", 25); // 實例化
			document.write(person.name);
			document.write(person.getProfile());
